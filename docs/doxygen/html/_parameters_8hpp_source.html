53inline auto

282 template<typename T>
283 OPTNONE auto hasNewValue(Key key, const T& v) const noexcept -> bool
284 {
285 if (!m_values.count(key)) {
286 KASSERT(false,
287 "Value is expected to exist here, implementation error (key "
288 "is invalid)");
289
290 return true;
291 }
292
293 const auto& currentValue = m_values.at(key);
294
295 // If we currently have no value, any value is new
296 if (!currentValue.hasValue()) {
297 return true;
298 }
299
300 // Otherwise, we expect types to match, otherwise this is an
301 // implementation error
302 if (!currentValue.isType<T>()) {
303 KASSERT(false, "Unexpected type, implementation error");
304
305 return false;
306 }
307
308 const auto& a = currentValue.getValue<T>();
309 const auto& b = v;
310
311 // Finally, check the current value
312 return a != b;
313 }

365 OPTNONE auto setValue(Key key, const T& v) noexcept -> bool
366 {

368

372 if (!validate(key, value)) {
373 return false;

378

380 m_values.emplace(key, std::move(value));
381
382 return true;
383 }

393 {

395 /* Spectrogram Parameters */

514 // Explicitly initialize values that may not always be set
